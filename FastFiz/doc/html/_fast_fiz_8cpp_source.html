94  {
95  //first, convert to [0,2*pi)
96  double twoPi = 2 * M_PI;
97  if(angle < 0)
98  {
99  double frac = angle / twoPi;
100  double residual = frac - ((int)frac);
101  angle = (1 + residual) * twoPi;
102  }
103  if(angle >= twoPi)
104  {
105  double frac = angle / twoPi;
106  double residual = frac - ((int)frac);
107  angle = residual * twoPi;
108  }
109 
110  if(START_NEG_HALF_CIRCLE == range)
111  {
112  if(angle >= M_PI)
113  {
114  angle -= twoPi;
115  }

850  void CueStrikeEvent::doHandle(TableState &ts, bool VERBOSE) const {
851  Ball &ball = ts.getBall(_ball1);
852 
853  double m = Ball::BALL_MASS;
854  double M = Table::CUE_MASS;
855  double cos = gsl_sf_cos(_params.theta * M_PI/180);
856  double sin = gsl_sf_sin(_params.theta * M_PI/180);
857  double c = fabs(sqrt(ball.getRadius()*ball.getRadius() - _params.a*_params.a - _params.b*_params.b));
858  //TODO: Check that c is sane
859 
860  //correctional constant to match poolfiz
861  double c1 = 0.999425;
862 
863  //F is the force of the cue stick on the ball
864  //Equation (1) in Leckie and Greenspan
865  double F = (2*m*_params.v) / (1 + (m/M) + c1*(5.0/(2.0*ball.getRadius()*ball.getRadius()))*(_params.a*_params.a + _params.b*_params.b*cos*cos + c*c*sin*sin - 2*_params.b*c*cos*sin));
866  if(VERBOSE)
867  printf("F = %g\n", F);
868  //Now, set the velocity of the cue ball
869  //In i,j,k coordinate system:
870  //velocity is (0, -F*cos/m, 0), making no-jumps assumption (2)
871  //angular velocity is (1/I)*(-c*F*sin + b*F*cos, a*F*sin, -a*F*cos) (3)
872 
873  //The following equations ((2) and (3) from Leckie/Greenspan) are in a
874  //ball-centric coordinate system (ijk). j is parallel to the projection of
875  //the cue onto the table; k is the same as z; and the ball's center is at
876  //(0, 0, 0). v_rotate returns this orientation to that of the table.
877  ball.setVelocity(Vector(0, -F*cos/m, 0).rotateDeg(_params.phi + 90.0));

879

881  //handleStateChange(ball);
882  ts.getBall(_ball1).updateState(VERBOSE);
883 
884  if(VERBOSE)
885  {
886  printf("Just struck the cue ball; cue ball has state:\n");
